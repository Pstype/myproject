// 创建一个画布
function setup() {
createCanvas(600, 600); // 创建一个600x600大小的画布
}

// 在画布上绘制内容
function draw() {
background(30); // 设置背景颜色为深色，以突显图案
noFill(); // 不填充形状
// 循环绘制多个图形，形成炫酷的图案
for (let i = 0; i < 360; i += 10) { // 每隔10度循环
stroke(map(i, 0, 360, 0, 255), 100, 200); // 根据角度设定颜色
strokeWeight(2); // 设置线条的粗细

// 计算圆形的位置和大小
let x = width / 2 + cos(radians(i)) * (200 + 50 * sin(radians(frameCount + i)));
let y = height / 2 + sin(radians(i)) * (200 + 50 * sin(radians(frameCount + i)));

// 绘制一个圆形
ellipse(x, y, 50 + 20 * sin(frameCount * 0.1 + radians(i)), 50 + 20 * sin(frameCount * 0.1 + radians(i)));
}
}

// 这段代码使用了 p5.js 库来创建一个动态炫酷的图案。
// 主要概念包括：
// 1. `setup()`: 在程序开始时运行，用于初始化设置，例如创建画布。
// 2. `draw()`: 每次画布需要重新绘制时运行，我们可以在这里绘制动画效果。
// 3. `for`循环: 用于重复执行代码，这里我们用它来绘制多个圆形。
// 4. `map()`: 用于将一个值转换成另一个范围的值，这里用于设置颜色。
// 5. `cos()` 和 `sin()`: 三角函数，可以帮助我们计算圆形的坐标，创造动态效果。
// 6. `radians()`: 将角度转换为弧度，这是因为三角函数在 p5.js 中使用的是弧度。
// 7. `frameCount`: 用于获得自程序开始以来运行的帧数，可以用于创建动画效果。
